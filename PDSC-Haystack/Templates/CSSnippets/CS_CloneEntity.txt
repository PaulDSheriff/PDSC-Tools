    #region Clone Entity Class
    /// <summary>
    /// Clones the current <|ENTITY_CLASS_NAME|>
    /// </summary>
    /// <returns>A cloned <|CLASS_NAME|> object</returns>
    public <|ENTITY_CLASS_NAME|> CloneEntity()
    {
      return CloneEntity(this.Entity);
    }
    
    /// <summary>
    /// Clones the passed in <|ENTITY_CLASS_NAME|>
    /// </summary>
    /// <param name="entityToClone">The <|ENTITY_CLASS_NAME|> entity to clone</param>
    /// <returns>A cloned <|ENTITY_CLASS_NAME|> object</returns>
    public <|ENTITY_CLASS_NAME|> CloneEntity(<|ENTITY_CLASS_NAME|> entityToClone)
    {
      <|ENTITY_CLASS_NAME|> newEntity = new <|ENTITY_CLASS_NAME|>();

      {|FOR EACH COLUMN|}
      newEntity.<|COLUMN_PUBLIC_PROPERTY|> = entityToClone.<|COLUMN_PUBLIC_PROPERTY|>;
      {|END|}

      return newEntity;
    }
    #endregion
